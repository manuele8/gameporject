def istogrammasterisco(lista):
    counter_while = 0
    counter_for = 0
    istogramma = ""
    while counter_while < len(lista):
        for counter_for in range(lista[counter_while]):
            istogramma = istogramma + "*"
            counter_for += 1
        print(istogramma)
        istogramma = ""
        counter_while += 1
istogrammasterisco([30000, 7, 10, 23, 5, 2, 3, 4, 5, 3])
